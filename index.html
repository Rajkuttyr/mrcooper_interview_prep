CREATE TABLE customer (
  cust_id    VARCHAR(20) PRIMARY KEY,
  name       VARCHAR(100) NOT NULL,
  dob        DATE,
  email      VARCHAR(100) UNIQUE,
  phone      VARCHAR(15),
  address    TEXT,
  kyc_status VARCHAR(20) DEFAULT 'PENDING',
  tier       VARCHAR(10) DEFAULT 'REGULAR',
  created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE branch (
  branch_id  VARCHAR(10) PRIMARY KEY,
  name       VARCHAR(100),
  ifsc       VARCHAR(11) UNIQUE,
  city       VARCHAR(50),
  state      VARCHAR(50)
);

CREATE TABLE account (
  acc_no      VARCHAR(20) PRIMARY KEY,
  cust_id     VARCHAR(20) REFERENCES customer(cust_id),
  branch_id   VARCHAR(10) REFERENCES branch(branch_id),
  acc_type    VARCHAR(20),  -- SAVINGS, CURRENT, FD
  balance     DECIMAL(15,2) DEFAULT 0.00,
  status      VARCHAR(10) DEFAULT 'ACTIVE',
  created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE transactions (
  txn_id      VARCHAR(30) PRIMARY KEY,
  from_acc    VARCHAR(20) REFERENCES account(acc_no),
  to_acc      VARCHAR(20) REFERENCES account(acc_no),
  amount      DECIMAL(15,2),
  txn_type    VARCHAR(20),  -- CREDIT, DEBIT, TRANSFER
  status      VARCHAR(10),
  txn_date    TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE loan (
  loan_id     VARCHAR(20) PRIMARY KEY,
  cust_id     VARCHAR(20) REFERENCES customer(cust_id),
  loan_type   VARCHAR(20),
  principal   DECIMAL(15,2),
  rate        DECIMAL(5,2),
  tenure_months INT,
  emi_amount  DECIMAL(15,2),
  status      VARCHAR(20) DEFAULT 'ACTIVE'
);

CREATE TABLE card (
  card_id     VARCHAR(20) PRIMARY KEY,
  acc_no      VARCHAR(20) REFERENCES account(acc_no),
  card_type   VARCHAR(10),
  masked_no   VARCHAR(20),
  expiry      DATE,
  credit_limit DECIMAL(15,2),
  status      VARCHAR(10) DEFAULT 'ACTIVE'
);

-- Q1: Customer account summary with branch (INNER JOIN)
SELECT c.name, c.email, a.acc_no, a.acc_type, a.balance, b.name AS branch
FROM customer c
JOIN account a ON c.cust_id = a.cust_id
JOIN branch b ON a.branch_id = b.branch_id
WHERE a.status = 'ACTIVE';

-- Q2: Total deposits & withdrawals per customer (GROUP BY + SUM)
SELECT a.cust_id, c.name,
  SUM(CASE WHEN t.txn_type='CREDIT' THEN t.amount ELSE 0 END) credits,
  SUM(CASE WHEN t.txn_type='DEBIT'  THEN t.amount ELSE 0 END) debits
FROM transactions t
JOIN account a ON t.from_acc = a.acc_no
JOIN customer c ON a.cust_id = c.cust_id
GROUP BY a.cust_id, c.name;

-- Q3: Top 5 high-value transactions (ORDER + LIMIT)
SELECT txn_id, from_acc, to_acc, amount, txn_date
FROM transactions
ORDER BY amount DESC
LIMIT 5;

-- Q4: Running balance using WINDOW FUNCTION
SELECT txn_id, amount, txn_type, txn_date,
  SUM(amount) OVER (
    PARTITION BY from_acc
    ORDER BY txn_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
  ) AS running_total
FROM transactions;

-- Q5: Rank customers by balance (RANK window function)
SELECT c.name, a.balance,
  RANK() OVER (ORDER BY a.balance DESC) AS wealth_rank,
  DENSE_RANK() OVER (ORDER BY a.balance DESC) AS dense_rank
FROM account a JOIN customer c ON a.cust_id = c.cust_id;

-- Q6: Customers with NO active loans (LEFT JOIN + IS NULL / subquery)
SELECT c.cust_id, c.name
FROM customer c
WHERE c.cust_id NOT IN (
  SELECT DISTINCT cust_id FROM loan WHERE status = 'ACTIVE'
);

-- Q7: Monthly transaction volume (DATE_TRUNC + GROUP BY)
SELECT
  DATE_TRUNC('month', txn_date) AS month,
  COUNT(*) AS txn_count,
  SUM(amount) AS total_volume
FROM transactions
GROUP BY 1
ORDER BY 1;

-- Q8: Find duplicate transactions (potential fraud) - SUBQUERY
SELECT from_acc, to_acc, amount, txn_date, COUNT(*)
FROM transactions
GROUP BY from_acc, to_acc, amount, txn_date
HAVING COUNT(*) > 1;

-- Q9: Average balance per branch (JOIN + AVG)
SELECT b.name, b.city, AVG(a.balance) AS avg_balance, COUNT(a.acc_no) AS total_accounts
FROM branch b
JOIN account a ON b.branch_id = a.branch_id
GROUP BY b.name, b.city
ORDER BY avg_balance DESC;

-- Q10: CTE — customers with balance above branch average
WITH branch_avg AS (
  SELECT branch_id, AVG(balance) AS avg_bal
  FROM account GROUP BY branch_id
)
SELECT c.name, a.balance, ba.avg_bal
FROM account a
JOIN branch_avg ba ON a.branch_id = ba.branch_id
JOIN customer c ON a.cust_id = c.cust_id
WHERE a.balance > ba.avg_bal;

-- Q11: LAG/LEAD — compare transaction with previous (WINDOW)
SELECT txn_id, amount, txn_date,
  LAG(amount) OVER (PARTITION BY from_acc ORDER BY txn_date) AS prev_txn,
  amount - LAG(amount) OVER (PARTITION BY from_acc ORDER BY txn_date) AS diff
FROM transactions;

-- Q12: Accounts with no transactions in 6 months (inactive)
SELECT acc_no, cust_id, balance
FROM account
WHERE acc_no NOT IN (
  SELECT DISTINCT from_acc FROM transactions
  WHERE txn_date >= NOW() - INTERVAL '6 months'
);

-- Q13: Loan repayment status with customer info (3-table JOIN)
SELECT c.name, l.loan_id, l.principal, l.emi_amount,
  COUNT(lp.pay_id) AS paid_emis
FROM loan l
JOIN customer c ON l.cust_id = c.cust_id
LEFT JOIN loan_payment lp ON l.loan_id = lp.loan_id
GROUP BY c.name, l.loan_id, l.principal, l.emi_amount;

-- Q14: NTILE — segment customers by balance quartile
SELECT c.name, a.balance,
  NTILE(4) OVER (ORDER BY a.balance) AS quartile
FROM account a JOIN customer c ON a.cust_id = c.cust_id;

-- Q15: Self-join: accounts sending money to each other
SELECT DISTINCT t1.from_acc, t1.to_acc
FROM transactions t1
JOIN transactions t2 ON t1.from_acc = t2.to_acc
  AND t1.to_acc = t2.from_acc;

-- Q16: UPDATE balance after transaction (DML)
BEGIN;
  UPDATE account SET balance = balance - 5000
    WHERE acc_no = 'ACC001' AND balance >= 5000;
  UPDATE account SET balance = balance + 5000
    WHERE acc_no = 'ACC002';
COMMIT;

-- Q17: FIRST_VALUE / LAST_VALUE per account (window)
SELECT from_acc, amount, txn_date,
  FIRST_VALUE(amount) OVER (PARTITION BY from_acc ORDER BY txn_date) AS first_txn,
  LAST_VALUE(amount)  OVER (PARTITION BY from_acc ORDER BY txn_date
    ROWS BETWEEN UNBOUNDED PRECEDING AND UNBOUNDED FOLLOWING) AS last_txn
FROM transactions;

-- Q18: Pivot-style: transaction counts by type per month
SELECT
  DATE_TRUNC('month', txn_date) AS month,
  SUM(CASE WHEN txn_type='CREDIT'   THEN 1 ELSE 0 END) credits,
  SUM(CASE WHEN txn_type='DEBIT'    THEN 1 ELSE 0 END) debits,
  SUM(CASE WHEN txn_type='TRANSFER' THEN 1 ELSE 0 END) transfers
FROM transactions GROUP BY 1;

-- Q19: Correlated subquery — customers above average balance
SELECT c.name, a.balance
FROM customer c JOIN account a ON c.cust_id = a.cust_id
WHERE a.balance > (SELECT AVG(balance) FROM account);

SELECT c.name, l.loan_id, l.loan_type, l.status
FROM customer c
FULL OUTER JOIN loan l ON c.cust_id = l.cust_id

// ===== ABSTRACT BASE CLASS =====
public abstract class Account {
    protected String accountId;
    protected double balance;
    protected Customer owner;
    protected List<Transaction> history = new ArrayList<>();

    public abstract double calculateInterest();

    public synchronized void deposit(double amount) {
        if (amount <= 0) throw new IllegalArgumentException("Amount must be positive");
        this.balance += amount;
        history.add(new Transaction(accountId, null, amount, "CREDIT"));
    }

    public synchronized boolean withdraw(double amount) {
        if (amount > balance) return false;
        this.balance -= amount;
        history.add(new Transaction(null, accountId, amount, "DEBIT"));
        return true;
    }
}

// ===== SAVINGS ACCOUNT =====
public class SavingsAccount extends Account {
    private static final double INTEREST_RATE = 0.04;

    public double calculateInterest() {
        return balance * INTEREST_RATE;
    }
}

// ===== BANK SERVICE (uses HashMap + PriorityQueue DSA) =====
public class BankService {
    // HashMap O(1) lookup by account number
    private Map<String, Account> accounts = new HashMap<>();
    // PriorityQueue for high-value transaction alerts
    private PriorityQueue<Transaction> alerts =
        new PriorityQueue<>(Comparator.comparingDouble(Transaction::getAmount).reversed());

    public boolean transfer(String from, String to, double amount) {
        Account src = accounts.get(from);
        Account dst = accounts.get(to);
        if (src == null || dst == null) return false;

        // Lock in consistent order to prevent deadlock
        Object lock1 = src.accountId.compareTo(dst.accountId) < 0 ? src : dst;
        Object lock2 = src.accountId.compareTo(dst.accountId) < 0 ? dst : src;

        synchronized(lock1) {
            synchronized(lock2) {
                if (!src.withdraw(amount)) return false;
                dst.deposit(amount);
                if (amount > 50000) alerts.offer(new Transaction(from, to, amount, "TRANSFER"));
                return true;
            }
        }
    }

    // DSA: Find top-K transactions using PriorityQueue O(n log k)
    public List<Transaction> getTopKTransactions(int k) {
        return IntStream.range(0, k)
            .filter(i -> !alerts.isEmpty())
            .mapToObj(i -> alerts.poll())
            .collect(Collectors.toList());
    }
}

// ===== INTERFACE + DESIGN PATTERN =====
public interface Notifiable {
    void sendAlert(String message);
}
// Observer Pattern for fraud detection
public class FraudDetectionService implements Notifiable {
    private Map<String, Deque<Transaction>> recentTxns = new HashMap<>();

    public boolean isSuspicious(Transaction t) {
        Deque<Transaction> recent = recentTxns
            .getOrDefault(t.getFromAcc(), new ArrayDeque<>());
        // Sliding window: 3 large txns in 1 min = fraud
        return recent.size() >= 3 &&
            recent.stream().allMatch(x -> x.getAmount() > 10000);
    }
    public void sendAlert(String message) {
        System.out.println("[FRAUD ALERT] " + message);
    }
}

CREATE TABLE users (
  user_id    BIGINT PRIMARY KEY AUTO_INCREMENT,
  name       VARCHAR(100),
  email      VARCHAR(100) UNIQUE NOT NULL,
  phone      VARCHAR(15),
  created_at DATETIME DEFAULT NOW()
);

CREATE TABLE category (
  cat_id     INT PRIMARY KEY AUTO_INCREMENT,
  name       VARCHAR(100),
  parent_id  INT REFERENCES category(cat_id)  -- self join
);

CREATE TABLE product (
  prod_id    BIGINT PRIMARY KEY AUTO_INCREMENT,
  name       VARCHAR(200),
  price      DECIMAL(10,2),
  stock      INT DEFAULT 0,
  cat_id     INT REFERENCES category(cat_id),
  seller_id  BIGINT,
  rating     DECIMAL(3,2) DEFAULT 0
);

CREATE TABLE orders (
  order_id   BIGINT PRIMARY KEY AUTO_INCREMENT,
  user_id    BIGINT REFERENCES users(user_id),
  total_amt  DECIMAL(10,2),
  status     VARCHAR(20) DEFAULT 'PLACED',
  created_at DATETIME DEFAULT NOW()
);

CREATE TABLE order_item (
  item_id    BIGINT PRIMARY KEY AUTO_INCREMENT,
  order_id   BIGINT REFERENCES orders(order_id),
  prod_id    BIGINT REFERENCES product(prod_id),
  qty        INT,
  price_snapshot DECIMAL(10,2)
);

CREATE TABLE review (
  review_id  BIGINT PRIMARY KEY AUTO_INCREMENT,
  user_id    BIGINT REFERENCES users(user_id),
  prod_id    BIGINT REFERENCES product(prod_id),
  rating     TINYINT CHECK(rating BETWEEN 1 AND 5),
  comment    TEXT
);

-- Q1: Top 10 best-selling products (JOIN + GROUP BY)
SELECT p.name, SUM(oi.qty) AS total_sold, SUM(oi.qty * oi.price_snapshot) revenue
FROM order_item oi JOIN product p ON oi.prod_id = p.prod_id
JOIN orders o ON oi.order_id = o.order_id WHERE o.status = 'DELIVERED'
GROUP BY p.prod_id, p.name ORDER BY total_sold DESC LIMIT 10;

-- Q2: Revenue per category (recursive category using CTE)
WITH RECURSIVE cat_tree AS (
  SELECT cat_id, name, parent_id FROM category WHERE parent_id IS NULL
  UNION ALL
  SELECT c.cat_id, c.name, c.parent_id FROM category c
  JOIN cat_tree ct ON c.parent_id = ct.cat_id
)
SELECT ct.name, SUM(oi.qty * oi.price_snapshot) revenue
FROM cat_tree ct JOIN product p ON ct.cat_id = p.cat_id
JOIN order_item oi ON p.prod_id = oi.prod_id GROUP BY ct.name;

-- Q3: Customers who never placed an order (LEFT JOIN)

// ===== CART (uses LinkedHashMap for insertion order) =====
public class ShoppingCart {
    private Map<Product, Integer> items = new LinkedHashMap<>();

    public void addItem(Product p, int qty) {
        items.merge(p, qty, Integer::sum);
    }
    public void removeItem(Product p) { items.remove(p); }
    public double getTotal() {
        return items.entrySet().stream()
            .mapToDouble(e -> e.getKey().getPrice() * e.getValue())
            .sum();
    }

// ===== PRODUCT SEARCH (Trie for autocomplete DSA) =====
class TrieNode {
    Map<Character, TrieNode> children = new HashMap<>();
    boolean isEnd;
    List<String> suggestions = new ArrayList<>();
}
public class ProductSearch {
    private TrieNode root = new TrieNode();

    public void insert(String product) {
        TrieNode node = root;
        for (char c : product.toCharArray()) {
            node = node.children.computeIfAbsent(c, k -> new TrieNode());
            node.suggestions.add(product);
        }
        node.isEnd = true;
    }

    public List<String> autocomplete(String prefix) {
        TrieNode node = root;
        for (char c : prefix.toCharArray()) {
            if (!node.children.containsKey(c)) return new ArrayList<>();
            node = node.children.get(c);
        }
        return node.suggestions;
    }
}

// ===== RECOMMENDATION (Graph BFS — "users who watched X also watched") =====
public class RecommendationEngine {
    // Adjacency list: video -> related videos
    private Map<String, List<String>> graph = new HashMap<>();

    public List<String> getRecommendations(String seed, int limit) {
        Queue<String> queue = new LinkedList<>();
        Set<String> visited = new HashSet<>();
        List<String> result = new ArrayList<>();

        queue.offer(seed); visited.add(seed);
        while (!queue.isEmpty() && result.size() < limit) {
            String curr = queue.poll();
            for (String neighbor : graph.getOrDefault(curr, List.of())) {
                if (visited.add(neighbor)) {
                    result.add(neighbor);
                    queue.offer(neighbor);
                }
            }
        }
        return result;
    }
}

// ===== VIEW COUNT (AtomicLong for thread-safety) =====
public class ViewCounter {
    private Map<String, AtomicLong> counts = new ConcurrentHashMap<>();

    public void increment(String videoId) {
        counts.computeIfAbsent(videoId, k -> new AtomicLong(0)).incrementAndGet();
    }
    public long getCount(String videoId) {
        return counts.getOrDefault(videoId, new AtomicLong(0)).get();
    }
}

// ===== DRIVER MATCHING (Greedy + PriorityQueue) =====
public class DriverMatchingService {
    // Min-heap by distance to order location
    public Optional<Driver> findNearest(double lat, double lng,
                                           List<Driver> available) {
        return available.stream()
            .filter(Driver::isAvailable)
            .min(Comparator.comparingDouble(d ->
                haversine(lat, lng, d.getLat(), d.getLng())));
    }
    // Haversine formula for geo distance
    private double haversine(double lat1, double lon1, double lat2, double lon2) {
        double R = 6371;
        double dLat = Math.toRadians(lat2 - lat1);
        double dLon = Math.toRadians(lon2 - lon1);
        double a = Math.sin(dLat/2) * Math.sin(dLat/2)
            + Math.cos(Math.toRadians(lat1)) * Math.cos(Math.toRadians(lat2))
            * Math.sin(dLon/2) * Math.sin(dLon/2);
        return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    }
}

// ===== ORDER STATE MACHINE (State Pattern) =====
public enum OrderStatus { PLACED, CONFIRMED, PREPARING, PICKED_UP, DELIVERED, CANCELLED }

public class Order {
    private OrderStatus status = OrderStatus.PLACED;
    private List<StatusChangeListener> listeners = new ArrayList<>();

    public void advance() {
        switch(status) {
            case PLACED:   status = OrderStatus.CONFIRMED;  break;
            case CONFIRMED: status = OrderStatus.PREPARING; break;
            case PREPARING: status = OrderStatus.PICKED_UP; break;
            case PICKED_UP: status = OrderStatus.DELIVERED; break;
            default: throw new IllegalStateException();
        }
        listeners.forEach(l -> l.onStatusChange(status));
    }
}

// ===== MESSAGE (Enum + Records) =====
public enum MessageType { TEXT, IMAGE, VIDEO, VOICE, FILE }
public enum MessageStatus { SENT, DELIVERED, READ }

public class Message {
    private final String messageId;
    private final String conversationId;
    private final String senderId;
    private final String content;
    private final MessageType type;
    private volatile MessageStatus status;
    private final long timestamp;
}

// ===== CHAT SERVICE (uses ConcurrentHashMap + LinkedList) =====
public class ChatService {
    // conversation_id -> ordered messages (LinkedList preserves order)
    private Map<String, LinkedList<Message>> conversations = new ConcurrentHashMap<>();
    // userId -> websocket session
    private Map<String, WebSocketSession> sessions = new ConcurrentHashMap<>();

    public void sendMessage(Message msg) {
        conversations
            .computeIfAbsent(msg.getConversationId(), k -> new LinkedList<>())
            .addLast(msg);

        // Deliver to all members in conversation
        getConversationMembers(msg.getConversationId()).forEach(memberId -> {
            WebSocketSession session = sessions.get(memberId);
            if (session != null && session.isOpen()) {
                session.send(msg); // real-time delivery
                msg.setStatus(MessageStatus.DELIVERED);
            }
        });
    }

    // DSA: Get last N messages efficiently
    public List<Message> getRecent(String convId, int n) {
        LinkedList<Message> msgs = conversations.getOrDefault(convId, new LinkedList<>());
        return msgs.stream().skip(Math.max(0, msgs.size()-n)).collect(Collectors.toList());
    }
}

// ===== ONLINE PRESENCE (using HashMap + TTL concept) =====
public class PresenceService {
    private Map<String, Long> lastSeen = new ConcurrentHashMap<>();
    private static final long ONLINE_THRESHOLD_MS = 5 * 60 * 1000; // 5 min

    public void heartbeat(String userId) {
        lastSeen.put(userId, System.currentTimeMillis());
    }
    public boolean isOnline(String userId) {
        Long ts = lastSeen.get(userId);
        return ts != null && (System.currentTimeMillis() - ts) < ONLINE_THRESHOLD_MS;
    }
}

// ===== BOOK CATALOG (uses HashMap + TreeMap) =====
public class Library {
    // O(1) lookup by ISBN
    private Map<String, Book> catalog = new HashMap<>();
    // TreeMap for sorted browsing by title
    private TreeMap<String, Book> byTitle  = new TreeMap<>();
    // Queue for reservation
    private Map<String, Queue<Member>> reservations = new HashMap<>();

    public Optional<Loan> issueBook(String isbn, Member member) {
        Book book = catalog.get(isbn);
        if (book == null || book.getAvailable() == 0) {
            addReservation(isbn, member);
            return Optional.empty();
        }
        if (member.hasReachedLoanLimit()) throw new LoanLimitException();
        book.decrementAvailable();
        return Optional.of(new Loan(isbn, member, LocalDate.now(),
            LocalDate.now().plusDays(14)));
    }

    public void returnBook(Loan loan) {
        Book book = catalog.get(loan.getIsbn());
        long overdueDays = ChronoUnit.DAYS.between(loan.getDueDate(), LocalDate.now());
        if (overdueDays > 0) loan.setFine(overdueDays * 2.0);

        book.incrementAvailable();
        loan.setStatus("RETURNED");

        // Notify next in reservation queue
        Queue<Member> queue = reservations.get(loan.getIsbn());
        if (queue != null && !queue.isEmpty()) {
            Member next = queue.poll();
            notifyMember(next, "Book is now available: " + book.getTitle());
        }
    }

    // DSA: Binary search on sorted title list
    public List<Book> searchByTitlePrefix(String prefix) {
        return new ArrayList<>(
            byTitle.subMap(prefix, prefix + Character.MAX_VALUE).values()
        );
    }
}
